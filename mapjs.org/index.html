MapReduce using V8 JavaScript and Node.js
/* Harnesses the prowess of V8 and the versatility of Node.js to setup MapReduce system with ease in JavaScript. */
var map = require('mapjs').map;
var reduce = require('mapjs').reduce;
var JobConf = require('mapjs').jobconf;
var JobClient = require('mapjs').jobclient;

var mapper = map.createMapper(function(key, value, outputCollector, reporter) {
    // process the key/value pair.
    outputCollector.collect(key, value);
});

var reducer = reduce.createReducer(function(key, values, outputCollector, reporter) {
    // process the key and the list of values.
    outputCollector.collect(key, value);
});

var jobconf = new JobConf();
jobconf.mapper = mapper;
jobconf.reducer = reducer;
jobconf.inputPath = "./";
jobconf.outputPath = "./output";
JobClient.run(jobconf);
